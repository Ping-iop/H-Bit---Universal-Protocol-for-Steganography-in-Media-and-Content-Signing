import numpy as np
from PIL import Image
import hashlib

def generate_author_hash(key_string):
    """
    Genera una huella digital única (SHA-256) basada en la clave del autor.
    """
    return hashlib.sha256(key_string.encode()).hexdigest()

def encode_hbit_redundant(image_path, secret_key, output_path, channel=2):
    """
    Incrusta la clave del autor repetidamente en toda la imagen para 
    resistir recortes (cropping).
    """
    # 1. Preparar la firma base
    signature = generate_author_hash(secret_key)
    binary_signature = ''.join(format(ord(c), '08b') for c in signature)
    # Marcador de inicio y fin para sincronización
    marker = '10101010' 
    full_unit = marker + binary_signature + marker
    
    # 2. Cargar imagen
    img = Image.open(image_path).convert('RGB')
    data = np.array(img)
    rows, cols, _ = data.shape
    
    total_pixels = rows * cols
    units_to_fill = total_pixels // len(full_unit)
    
    # Creamos la cadena larga que se repetirá en toda la imagen
    long_signature = full_unit * units_to_fill
    
    # 3. Proceso de incrustación
    idx = 0
    flat_data = data[:, :, channel].flatten() # Trabajamos en plano para velocidad
    
    # Solo iteramos hasta donde quepa nuestra firma repetida
    limit = min(len(long_signature), len(flat_data))
    
    for i in range(limit):
        bit = int(long_signature[i])
        # Aplicamos LSB (Bit menos significativo)
        flat_data[i] = (flat_data[i] & ~1) | bit
        
    # Reconstruimos la imagen
    data[:, :, channel] = flat_data.reshape((rows, cols))
    
    new_img = Image.fromarray(data)
    new_img.save(output_path)
    print(f"Éxito: Firma redundante incrustada {units_to_fill} veces.")

def decode_hbit_redundant(image_path, channel=2):
    """
    Extrae la firma analizando patrones repetidos. 
    Funciona incluso si la imagen ha sido recortada.
    """
    img = Image.open(image_path).convert('RGB')
    data = np.array(img)
    flat_data = data[:, :, channel].flatten()
    
    # Extraemos todos los LSB
    bits = "".join([str(b & 1) for b in flat_data])
    
    # Buscamos el patrón entre marcadores
    marker = '10101010'
    start = bits.find(marker)
    if start == -1:
        return "No se encontró firma."
        
    # Saltamos el marcador y buscamos el siguiente
    content_start = start + len(marker)
    end = bits.find(marker, content_start)
    
    if end == -1:
        return "Firma corrupta o incompleta."
        
    binary_hash = bits[content_start:end]
    
    # Convertimos de binario a texto
    try:
        chars = [binary_hash[i:i+8] for i in range(0, len(binary_hash), 8)]
        extracted_hash = "".join([chr(int(char, 2)) for char in chars])
        return extracted_hash
    except:
        return "Error en la decodificación del fragmento."
